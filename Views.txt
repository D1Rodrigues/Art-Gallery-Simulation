modern_works (artwork_name, art_number, artist_name, period) AS (SELECT art_Identity.name, art_Identity.pieceId, artist.Name, artist.periodId FROM art_Identity, artist WHERE artist.periodId >= 7 AND artist.artistid = art_identity.artistid)
admin_team_roles (user_num, user_role, admin_num) AS SELECT admin.userID, admin_role.role, admin.adminID FROM admin, admin_role WHERE admin.adminId = admin_role.adminID and admin_role.role = 'Mod' OR admin_role.role = 'Admin' ORDER BY admin.adminId ASC
Ancient_Vs_Contemp_Artists (piece, name, period) AS (Select art_identity.name As Art_Piece, artist.name As Artist, Period.name AS Worked_in From art_identity Join artist ON art_identity.artistid = artist.artistid Join period ON artist.periodid = period.periodid AND (period.periodid = '1' OR period.periodid = '8')) 
Art_On_Display (Name, Artist) As ( Select art_identity.name, artist.name From art_identity, artist Where Exists( Select * From Display, contains Where( Display.GalleryId = '1' and ( Display.displayId = contains.displayId and Art_identity.pieceId = Contains.pieceId) and art_identity.artistid = artist.artistid)))
